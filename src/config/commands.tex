\newcommand{\cvSection}[1]
{
    \vspace{\cvVerticalSeparator}

    \noindent
    {\Large\bfseries #1}

    \vspace{0.2em}

    \hrule

    \vspace{\cvVerticalSeparator}
}

\newcommand{\cvName}[1]
{
    \begin{center}
        {\Huge\bfseries #1}
    \end{center}
}

\newcommand{\cvSubtitle}[1]
{
    \begin{center}
        {\large #1}
    \end{center}

    \vspace{\cvVerticalSeparator}
}



% \cvContact {Top Left} {Bottom Left} {Top Right} {Bottom Right}
\newcommand{\cvContact}[4]
{
    \begin{center}
        \begin{tabular*}{0.90\textwidth}{@{\extracolsep{\fill}}ll}
            #1 & #3 \\
            #2 & #4
        \end{tabular*}
    \end{center}

    \vspace{\cvVerticalSeparator}
}

\newcommand{\cvSummary}[1]
{
    \begin{itemize}[leftmargin=\cvLeftMargin,nosep]
    \cvItem[] #1
    \end{itemize}

    \vspace{\cvVerticalSeparator}
}

\newcommand{\cvItem}[1]

}
